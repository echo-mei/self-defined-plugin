分组选择demo(支持二级、三级数据)


    $("#gs-duoxuan").gruopSelect({
        dataUrl:"groupSelect/groupselect.json",
        wrapStyle:{
            top:"-150px",
            left:"105px",
            width:"800px",
            height:"370px"
        },
        onClickSure: function() {
            alert("你选中的是"+this.getSelectedItems());
        }
    });
        

    $("#gs-danxuan").gruopSelect({

            width:"700px",

        },
        onClickSure: function() {
            alert("你选中的是"+this.getSelectedItems());
        }
    });
        

    $("#gs-modal").gruopSelect({
